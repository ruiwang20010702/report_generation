function analyzePauses(
  utterances: Array<{ start: number; end: number; speaker?: string }>,
  speaker?: string

    ? utterances.filter(u => u.speaker === speaker)

      longPauses: 0,
      pauseFrequency: 0,

function calculateResponseSpeed(
  utterances: Array<{ start: number; end: number; speaker: string }>
): ResponseSpeed {
  const teacherUtterances = utterances.filter(u => u.speaker === 'Speaker 0');
  const studentUtterances = utterances.filter(u => u.speaker === 'Speaker 1');
  
  const responseTimes: number[] = [];

      averageResponseTime: 0,
      fastestResponse: 0,
      slowestResponse: 0,

    averageResponseTime: responseTimes.reduce((a, b) => a + b, 0) / responseTimes.length,
    fastestResponse: Math.min(...responseTimes),
    slowestResponse: Math.max(...responseTimes),
  };

function calculateSpeakingTimeRatio(
  utterances: Array<{ start: number; end: number; speaker: string }>
): SpeakingTimeRatio {

  const teacherUtterances = utterances.filter(u => u.speaker === 'Speaker 0');

      totalTime: 0,
      ratio: 0,
      studentUtteranceCount: 0,
      teacherUtteranceCount: 0,

    (min, u) => u.start < min.start ? u : min

    studentSpeakingTime,
    totalTime,

    studentUtteranceCount: studentUtterances.length,
    teacherUtteranceCount: teacherUtterances.length,
  };
}
